This package is free for use, see below a sample use:

	var p = new Push(
                new PushTokens
                {
                    OnesignalAppId = "The application Id",
                    RestApiKey = "Your Rest Api Key"
                }
	);

 	var destinatarios = new List<string>
	{
		"SOME_DEVICE_ID"
	};

	p.CreateWebPushMessage(destinatarios, "Message Title", "Message Content");
